class Solution 
{
    //Function to reverse words in a given string.
    String reverseWords(String S)
    {
        // code here 
    String[] parts = S.split("\\.");
    int n= parts.length;
    
    String str="";
    
    for(int i= n-1 ;i>=0 ;i--)
    {
        str+=parts[i];
        if(i != 0) str+=".";
    }
        
       
       return str; 
    }
}
